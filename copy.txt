const db = require("../config/db");

const User = {
  // L·∫•y to√†n b·ªô ng∆∞·ªùi d√πng
  async getAll() {
    return db.execute("SELECT * FROM users");
  },

  // T·∫°o ng∆∞·ªùi d√πng m·ªõi
  async create({ username, email, password }) {
    return db.execute(
      "INSERT INTO users (username, email, password, role) VALUES (?, ?, ?, 'user')",
      [username, email, password]
    );
  },

  // C·∫≠p nh·∫≠t ng∆∞·ªùi d√πng
  async update(id, { username, email, password, role }) {
    return db.execute(
      "UPDATE users SET username=?, email=?, password=?, role=? WHERE id=?",
      [username, email, password, role, id]
    );
  },

  // T√¨m ng∆∞·ªùi d√πng theo email
  async findByEmail(email) {
    const [rows] = await db.execute("SELECT * FROM users WHERE email = ?", [
      email,
    ]);
    return rows[0];
  },

  // Ki·ªÉm tra email c√≥ t·ªìn t·∫°i
  async isEmailExist(email) {
    const [rows] = await db.execute(
      "SELECT 1 FROM users WHERE email = ? LIMIT 1",
      [email]
    );
    return rows.length > 0;
  },

  // T√¨m ng∆∞·ªùi d√πng theo ID
  async findById(id) {
    const [rows] = await db.execute("SELECT * FROM users WHERE id = ?", [id]);
    return rows[0];
  },

  // Xo√° ng∆∞·ªùi d√πng
  async remove(id) {
    return db.execute("DELETE FROM users WHERE id=?", [id]);
  },

  // L∆∞u m√£ x√°c th·ª±c (n·∫øu b·∫°n mu·ªën l∆∞u v√†o DB, n√™n c√≥ b·∫£ng reset_codes ri√™ng)
  async saveResetCode(email, code) {
    // Ch·ªâ v√≠ d·ª•: b·∫°n c·∫ßn c√≥ b·∫£ng `reset_codes (email, code, expires_at)`
    return db.execute(
      "REPLACE INTO reset_codes (email, code, expires_at) VALUES (?, ?, DATE_ADD(NOW(), INTERVAL 15 MINUTE))",
      [email, code]
    );
  },

  async insertResetCode(email, code, expiresAt) {
    // Xo√° m√£ c≈© c√πng email (n·∫øu c√≥)
    await db.execute("DELETE FROM password_reset_codes WHERE email = ?", [
      email,
    ]);

    // Th√™m m√£ m·ªõi
    return db.execute(
      "INSERT INTO password_reset_codes (email, code, expires_at) VALUES (?, ?, ?)",
      [email, code, expiresAt]
    );
  },

  async verifyResetCode(email, code) {
    const [rows] = await db.execute(
      `SELECT * FROM password_reset_codes
     WHERE email = ? AND code = ? AND expires_at > NOW()
     ORDER BY created_at DESC
     LIMIT 1`,
      [email, code]
    );
    return rows[0];
  },
  async deleteResetCode(email) {
    return db.execute("DELETE FROM password_reset_codes WHERE email = ?", [
      email,
    ]);
  },

  // X√≥a c√°c m√£ h·∫øt h·∫°n
  async deleteResetCodeExpired() {
    return db.execute(
      "DELETE FROM password_reset_codes WHERE expires_at < NOW()"
    );
  },

  // ƒê·ªïi m·∫≠t kh·∫©u (s·ª≠ d·ª•ng khi m√£ x√°c th·ª±c h·ª£p l·ªá)
  async updatePasswordByEmail(email, newPassword) {
    return db.execute("UPDATE users SET password = ? WHERE email = ?", [
      newPassword,
      email,
    ]);
  },
};

exports.login = async (req, res) => {
  const { email, password } = req.body;
  try {
    const user = await User.findByEmail(email);
    if (!user)
      return res.status(401).json({ message: "T√†i kho·∫£n kh√¥ng t·ªìn t·∫°i" });

    const isMatch = await bcrypt.compare(password, user.password);
    if (!isMatch) return res.status(401).json({ message: "Sai m·∫≠t kh·∫©u" });

    const token = jwt.sign(
      { id: user.id, role: user.role },
      "your_secret_key",
      { expiresIn: "1d" }
    );

    res.json({
      token,
      user: { id: user.id, username: user.username, role: user.role },
    });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
};
